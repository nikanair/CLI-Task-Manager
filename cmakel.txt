cmake_minimum_required(VERSION 3.10)
project(TaskFlow VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(taskflow
    src/main.cpp
    src/Task.cpp
    src/TaskManager.cpp
    src/Utils.cpp
)

# Use std::filesystem on older compilers if necessary
if (MSVC)
  target_compile_options(taskflow PRIVATE /W4 /permissive-)
else()
  target_compile_options(taskflow PRIVATE -Wall -Wextra -Wpedantic -Wconversion)
endif()
